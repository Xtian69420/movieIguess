Usage Examples:

// Simple usage
showSuccess('Movie saved successfully!');
showError('Failed to load data');
showWarning('Check your connection');
showInfo('Welcome to MovieIGuess!');

// Advanced usage with options
showSuccess('Movie added!', {
    title: 'Success',
    duration: 5000,
    className: 'toast-large',
    showProgress: false,
    clickToDismiss: true
});

// Using the Toast instance directly
Toast.show('Custom message', 'success', 3000, {
    title: 'Custom Title'
});

// Remove all toasts
Toast.removeAll();
